const AttachMapRef = ({ mapRef, onReady }) => {
  const map = useMap();
 
  useEffect(() => {
    if (map && mapRef) {
      mapRef.current = map;
      onReady?.();
    }
  }, [map]);
 
  return null;
};
 
const RecenterMap = ({ center }) => {
  const map = useMap();
  useEffect(() => {
    if (center) {
      map.setView(center, map.getZoom());
    }
  }, [center]);
  return null;
};
